614  template<class T, std::size_t N>
615  struct euclidean_distance
616  {
617  static_assert(std::is_arithmetic<T>(), "first template parameter must be an arithemetic type");
618 
619  using vtraits = vector_traits<T, N>;
620 
621  typename vtraits::scalar_type
622  operator()(typename vtraits::param_type a, typename vtraits::param_type b)
623  {
624  typename vtraits::scalar_type sum = static_cast<typename vtraits::scalar_type>(0);
625  for (std::size_t i = 0; i < vtraits::cardinality; ++i)
626  {
627  typename vtraits::scalar_type diff = b[i] - a[i];
628  sum += diff * diff;
629  }
630  return static_cast<typename vtraits::scalar_type>( sqrt( static_cast<long double>(sum)));
631  }
632  };

977  gravitate()
978  {
979  bool new_results = false;
980  bool something_moved;
981  do
982  {
983  bool something_moved = false;
984 
985  for (std::size_t iclust = 0; iclust < current_clusters_->size(); ++iclust)
986  {

989  {

991  scalar_type max_grav{0};
992  std::size_t max_grav_cluster{iclust};

1000  #endif
1001  if (away_gravity > home_gravity)
1002  {
1003  if (away_gravity > max_grav)
1004  {
1005  max_grav = away_gravity;
1006  max_grav_cluster = jclust;
1007  }
1008  }
1009  }
1010  }
1011  if (max_grav_cluster != iclust)
1012  {
1013  something_moved = true;
1014  #if (TRACE)
1015  std::cout << "gravity moving index " << *it << " from cluster " << iclust << " to cluster " << max_grav_cluster << std::endl;
1016  #endif

1019  (*current_clusters_)[max_grav_cluster].members_.insert(moved_index);

1025  }
